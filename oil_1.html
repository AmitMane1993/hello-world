def show_result(y_test,X_test,y_test_predict,name,y_test_mse,test_accuracy,y_train_mse,train_accuracy):
  ## scattor plot
  plt.figure(figsize=(8,6))
  plt.scatter(y_test_predict[name],y_test,cmap='plasma')
  plt.title(name)
  plt.show()
  print('TRAIN RMSE for', name,' is ',y_train_mse[name], '\nand  R² is ',train_accuracy[name], '\n\nTEST RMSE for', name,' is ',y_test_mse[name], '\nand  R² is ',test_accuracy[name])
